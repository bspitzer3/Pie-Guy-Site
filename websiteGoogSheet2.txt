const SHEET_ID = 'PASTE_YOUR_GOOGLE_SHEET_ID_HERE';
const INVENTORY_SHEET = 'Inventory';
const ORDERS_SHEET = 'Orders';

function doGet() {
  return HtmlService.createHtmlOutputFromFile('Index')
    .setTitle('The Pie Guy | Limited Pie Orders');
}

function getSpreadsheet_() {
  return SpreadsheetApp.openById(SHEET_ID);
}

function getSheet_(name) {
  const sheet = getSpreadsheet_().getSheetByName(name);
  if (!sheet) {
    throw new Error(`Sheet not found: ${name}`);
  }
  return sheet;
}

function normalizeBoolean_(value) {
  if (value === true) return true;
  if (typeof value === 'string') {
    return value.toLowerCase() === 'true';
  }
  return false;
}

function getInventoryRecords_() {
  const sheet = getSheet_(INVENTORY_SHEET);
  const values = sheet.getDataRange().getValues();

  if (values.length < 2) {
    return [];
  }

  const headers = values[0].map(String);
  const rows = values.slice(1);

  return rows
    .map((row, index) => {
      const record = Object.fromEntries(headers.map((header, i) => [header, row[i]]));
      return {
        rowNumber: index + 2,
        sku: String(record.sku || '').trim(),
        name: String(record.name || '').trim(),
        price: Number(record.price || 0),
        remaining: Number(record.remaining || 0),
        active: normalizeBoolean_(record.active),
        updated_at: record.updated_at || '',
      };
    })
    .filter((item) => item.sku && item.active);
}

function buildPublicState_() {
  const items = getInventoryRecords_().map((item) => ({
    sku: item.sku,
    name: item.name,
    price: item.price,
    remaining: item.remaining,
  }));

  const totalRemaining = items.reduce((sum, item) => sum + item.remaining, 0);

  return {
    items,
    totalRemaining,
    isSoldOut: totalRemaining <= 0,
    serverTime: new Date().toISOString(),
  };
}

function getPublicState() {
  return buildPublicState_();
}

function submitOrder(order) {
  const lock = LockService.getScriptLock();
  lock.waitLock(30000);

  try {
    const name = String(order?.name || '').trim();
    const email = String(order?.email || '').trim();
    const phone = String(order?.phone || '').trim();
    const pickupDate = String(order?.pickupDate || '').trim();
    const notes = String(order?.notes || '').trim();
    const sku = String(order?.sku || '').trim();
    const quantity = Number(order?.quantity || 0);

    if (!name) throw new Error('Please enter your name.');
    if (!email) throw new Error('Please enter your email.');
    if (!sku) throw new Error('Please choose a pie.');
    if (!Number.isInteger(quantity) || quantity < 1) {
      throw new Error('Quantity must be at least 1.');
    }

    const inventorySheet = getSheet_(INVENTORY_SHEET);
    const ordersSheet = getSheet_(ORDERS_SHEET);
    const inventory = getInventoryRecords_();
    const item = inventory.find((entry) => entry.sku === sku);

    if (!item) {
      throw new Error('That pie is not currently available.');
    }

    if (item.remaining < quantity) {
      throw new Error(`Only ${item.remaining} ${item.remaining === 1 ? 'pie is' : 'pies are'} left for ${item.name}.`);
    }

    const newRemaining = item.remaining - quantity;
    const now = new Date();
    const orderId = `PG-${Utilities.formatDate(now, Session.getScriptTimeZone(), 'yyyyMMdd-HHmmss')}-${Math.floor(Math.random() * 1000)}`;

    // Update inventory first while lock is held.
    inventorySheet.getRange(item.rowNumber, 4).setValue(newRemaining); // remaining
    inventorySheet.getRange(item.rowNumber, 6).setValue(now);          // updated_at

    // Append immutable order log row.
    ordersSheet.appendRow([
      orderId,
      now,
      item.sku,
      item.name,
      quantity,
      item.price,
      name,
      email,
      phone,
      pickupDate,
      notes,
      newRemaining,
      'RESERVED',
    ]);

    SpreadsheetApp.flush();

    return {
      ok: true,
      message: `Thanks, ${name}! Your request for ${quantity} ${quantity === 1 ? 'pie' : 'pies'} has been captured for ${item.name}.`,
      orderId,
      state: buildPublicState_(),
    };
  } catch (error) {
    return {
      ok: false,
      message: error.message || 'Something went wrong while placing the order.',
      state: buildPublicState_(),
    };
  } finally {
    lock.releaseLock();
  }
}

function setupSheets() {
  const ss = getSpreadsheet_();

  let inventory = ss.getSheetByName(INVENTORY_SHEET);
  if (!inventory) inventory = ss.insertSheet(INVENTORY_SHEET);

  let orders = ss.getSheetByName(ORDERS_SHEET);
  if (!orders) orders = ss.insertSheet(ORDERS_SHEET);

  inventory.clear();
  orders.clear();

  inventory.getRange(1, 1, 1, 6).setValues([[
    'sku', 'name', 'price', 'remaining', 'active', 'updated_at'
  ]]);

  inventory.getRange(2, 1, 3, 6).setValues([
    ['apple', 'Apple Pie', 18, 12, true, ''],
    ['pumpkin', 'Pumpkin Pie', 20, 10, true, ''],
    ['cherry', 'Cherry Pie', 22, 8, true, ''],
  ]);

  orders.getRange(1, 1, 1, 13).setValues([[
    'order_id', 'created_at', 'sku', 'item_name', 'quantity', 'unit_price',
    'customer_name', 'email', 'phone', 'pickup_date', 'notes', 'remaining_after', 'status'
  ]]);
}

        orderStatusText.classList.add('status-open');
        soldOutPanel.style.display = 'none';
        orderForm.style.display = 'grid';
      }
    }

    function loadState(showError = false) {
      google.script.run
        .withSuccessHandler((state) => {
          renderState(state);
          if (!showError) return;
          setAlert();
        })
        .withFailureHandler((error) => {
          if (showError) {
            setAlert('error', error.message || 'Could not load current inventory.');
          }
        })
        .getPublicState();
    }

    skuSelect.addEventListener('change', () => {
      setAlert();
      updateSelectedAvailability();
    });

    quantityInput.addEventListener('input', () => {
      setAlert();
      updateSelectedAvailability();
    });

    orderForm.addEventListener('submit', (event) => {
      event.preventDefault();
      setAlert();

      const formData = new FormData(orderForm);
      const payload = {
        name: String(formData.get('name') || '').trim(),
        email: String(formData.get('email') || '').trim(),
        phone: String(formData.get('phone') || '').trim(),
        pickupDate: String(formData.get('pickupDate') || '').trim(),
        notes: String(formData.get('notes') || '').trim(),
        sku: String(formData.get('sku') || '').trim(),
        quantity: Number(formData.get('quantity') || 0),
      };

      submitButton.disabled = true;
      submitButton.textContent = 'Submitting...';

      google.script.run
        .withSuccessHandler((result) => {
          submitButton.disabled = false;
          submitButton.textContent = 'Reserve pies';
          renderState(result.state);
          setAlert(result.ok ? 'success' : 'error', result.message);
          if (result.ok) {
            orderForm.reset();
            quantityInput.value = 1;
            updateSelectedAvailability();
          }
        })
        .withFailureHandler((error) => {
          submitButton.disabled = false;
          submitButton.textContent = 'Reserve pies';
          setAlert('error', error.message || 'Could not submit your order.');
          loadState(false);
        })
        .submitOrder(payload);
    });

    loadState(true);
    setInterval(() => loadState(false), pollMs);
  </script>
